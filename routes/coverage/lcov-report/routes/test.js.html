var assert = require('assert');
var superagent = require('superagent');
var User = require('../models/user').User; 
var status = require('http-status');
 
describe('/users', function() {
  var app;
 
  before(function() {
    app = server(3000);
  });
 
  after(function() {
    app.close();
  });
 
  it('returns username if id param is a valid user', function(done) {
    superagent.get('http://localhost:3000/users/56dea2badad57fae043cc045').end(function(err, res) {
      assert.ifError(err);
      assert.equal(res.status, status.OK);
      var result = JSON.parse(res.text);
      assert.deepEqual({ user: '56dea2badad57fae043cc045' }, result);
      done();
    });
  });
 
  it('returns 404 if user named `params.id` not found', function(done) {
    superagent.get('http://localhost:3000/users/notfound').end(function(err, res) {
      assert.ifError(err);
      assert.equal(res.status, status.NOT_FOUND);
      var result = JSON.parse(res.text);
      assert.deepEqual({ error: 'Not Found' }, result);
      done();
    });
  });
});
